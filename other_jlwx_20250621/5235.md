第 13 题 下⾯程序的时间复杂度为（ ）。

    bool notPrime[N] = {false};
    void sieve() {
        for (int n = 2; n * n < N; n++)
            if (!notPrime[n])
                for (int i = n * n; i < N; i += n)
                    notPrime[i] = true;
    }
    

{{ select(13) }}

*   $O(N)$
*   $O(N \times log N)$
*   $O(N \times log log N)$
*   $O(N^2)$
