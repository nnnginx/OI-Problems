16. （阅读程序）

```
#include<cstdio>
#include<cstring>
using namespace std;
char st[100];
int main()
{
    scanf("%s",st);
    int n = strlen(st);
    for(int i = 1; i <= n; ++i) //第8行
    {
        if(n % i == 0)
        {
            char c = st[i - 1];
            if(c >= 'a')
                st[i - 1] = c - 'a' + 'A';
        }
    }
    printf("%s", st);
    return 0;
}
```
